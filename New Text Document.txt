void CentreGlassLR(float* valueLR)
{
	MeasureADCLR(valueLR);
	while(*valueLR>(RefrencePositionLeftRight + Hys) || *valueLR<(RefrencePositionLeftRight - Hys))
	{
		if(*valueLR<(RefrencePositionLeftRight))
		{
			Move_RIGHT();
		}
		else
		{
			Move_LEFT();
		}
			MeasureADCLR(valueLR);
	}
}
void CentreGlassUD(float* valueUD)
{
	MeasureADCUD(valueUD);
	while(*valueUD>(RefrencePositionUpDown + Hys) || *valueUD<(RefrencePositionUpDown - Hys))
	{

		{
			Move_UP();
		}
		else
		{
			Move_DOWN();
		}
		MeasureADCUD(valueUD);
	}
}
